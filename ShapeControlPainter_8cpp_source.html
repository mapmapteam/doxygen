59 void EllipseControlPainter::_paintShape(QPainter *painter)

62  Q_ASSERT(ellipse);
63 
64  // Init colors and stroke.
65  painter->setPen(_shapeItem->getRescaledShapeStroke());
66  painter->setBrush(Qt::NoBrush);
67 
68  // Draw ellipse contour.
69  QPainterPath path;
70  QTransform transform;
71  transform.translate(ellipse->getCenter().x(), ellipse->getCenter().y());
72  transform.rotate(ellipse->getRotation());
73  path.addEllipse(QPoint(0,0), ellipse->getHorizontalRadius(), ellipse->getVerticalRadius());
74  painter->drawPath(transform.map(path));
