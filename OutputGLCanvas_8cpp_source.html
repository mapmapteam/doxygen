65 void OutputGLCanvas::_drawTestSignal(QPainter* painter)
66 {
67  const QRect& geo = geometry();
68  painter->setPen(MM::CONTROL_COLOR);
69  int rect_size = 10;
70  QColor color_0(191, 191, 191);
71  QColor color_1(128, 128, 128);
72  QBrush brush_0(color_0);
73  QBrush brush_1(color_1);
74 
75  painter->setPen(Qt::NoPen);
76 
77  // Draw checkerboard pattern.
78  for (int x = geo.x(); x < geo.width(); x += rect_size)
79  {
80  for (int y = geo.y(); y < geo.height(); y += rect_size)
81  {
82  if (((x + y) % 20) == 0)
83  {
84  painter->setBrush(brush_0);
85  } else {
86  painter->setBrush(brush_1);
87  }
88  painter->drawRect(x, y, rect_size, rect_size);
89  }
90  }
91 
92  // Create responsive image
93  QImage test_signal = _svg_test_signal.scaled(geo.height(), geo.height(), Qt::KeepAspectRatio);
94  // Set new brush
95  this->_brush_test_signal = QBrush(test_signal);
96  // Center the brush
97  painter->translate((geo.width() - test_signal.width()) / 2, (geo.height() - test_signal.height()) / 2);
98  // Draw the actual brush.
99  painter->fillRect(0, 0, test_signal.width(), test_signal.height(), this->_brush_test_signal);
100 }
