50 void Polygon::_constrainVertex(const QPolygonF& polygon, int i, QPointF& v)
51 {
52  // Weird, but nothing to do.
53  if (polygon.size() <= 3)
54  return;
55 
56  // Save previous position of vertex.
57  QPointF prevV = polygon.at(i);
58 
59  // Look at the two adjunct segments to vertex i and see if they
60  // intersect with any non-adjacent segments.
61 
62  // Construct the list of segments (with the new candidate vertex).
63  QVector<QLineF> segments = _getSegments(polygon);
64  int prev = wrapAround(i - 1, segments.size());
65  int next = wrapAround(i + 1, segments.size());
66  segments[prev] = QLineF(polygon.at(prev), v);
67  segments[i] = QLineF(v, polygon.at(next));
68 
69  // We now stretch segments a little bit to cope with approximation errors.
70  for (QVector<QLineF>::Iterator it = segments.begin(); it != segments.end(); ++it)
71  {
72  QLineF& seg = *it;
73  QPointF p1 = seg.p1();
74  QPointF p2 = seg.p2();
75  seg.setP1( p1 + (p1 - p2) * 0.35f);
76  seg.setP2( p2 + (p2 - p1) * 0.35f);
77  }
78 
79  // For each adjunct segment.
80  for (int adj=0; adj<2; adj++)
81  {
82  int idx = wrapAround(i + adj - 1, segments.size());
83  for (int j=0; j<segments.size(); j++)
84  {
85  // If the segment to compare to is valid (ie. if it is not
86  // the segment itself nor an adjacent one) then check for
87  // intersection.
88  if (j != idx &&
89  j != wrapAround(idx-1, segments.size()) &&
90  j != wrapAround(idx+1, segments.size()))
91  {
92  QPointF intersection;
93  if (segments[idx].intersect(segments[j], &intersection) == QLineF::BoundedIntersection)
94  {
95  // Rearrange segments with new position at intersection point.
96  v = intersection;
97  segments[prev] = QLineF(polygon.at(prev), v);
98  segments[i] = QLineF(v, polygon.at(next));
99  }
100  }
101  }
102  }
103 }

255 void Mesh::addColumn()
256 {
257  // Create new vertices 2d (temporary).
258  IndexVector2d newVertices2d;
259  resizeVertices2d(newVertices2d, nColumns()+1, nRows());
260 
261  // Left displacement of points already there.
262  qreal leftMoveProp = 1.0f/(nColumns()-1) - 1.0f/nColumns();
263 
264  // Add a point at each row.
265  int k = nVertices();
266  for (int y=0; y<nRows(); y++)
267  {
268  // Get left and right vertices.
269  QPointF left = getVertex2d( 0, y );
270  QPointF right = getVertex2d( nColumns()-1, y );
271  QPointF diff = right - left;
272 
273  // First pass: move middle points.
274  for (int x=1; x<nColumns()-1; x++)
275  {
276  QPointF p = getVertex2d(x, y);
277  p -= diff * x * leftMoveProp;
278  _rawSetVertex( _vertices2d[x][y], p );
279  }
280 
281  // Create and add new point.
282  QPointF newPoint = right - diff * 1.0f/nColumns();
283  _addVertex(newPoint);
284 
285  // Assign new vertices 2d.
286  for (int x=0; x<nColumns()-1; x++)
287  newVertices2d[x][y] = _vertices2d[x][y];
288 
289  // The new point.
290  newVertices2d[nColumns()-1][y] = k;
291 
292  // The rightmost point.
293  newVertices2d[nColumns()][y] = _vertices2d[nColumns()-1][y];
294 
295  k++;
296  }
297 
298  // Copy new mapping.
299  _vertices2d = newVertices2d;
300 
301  // Increment number of columns.
302  _nColumns++;
303 
304  // Reorder.
305  _reorderVertices();

361 void Mesh::removeColumn(int columnId)
362 {
363  // Cannot remove first and last columns
364  Q_ASSERT(columnId >= 1 && columnId < nColumns()-1);
365 
366  // Temporary containers that will be used to rebuild new vertex space.
367  IndexVector2d newVertices2d;
368  resizeVertices2d(newVertices2d, nColumns()-1, nRows());
369 
370  QVector<QPointF> newVertices(vertices.size()-nRows());
371 
372  // Right displacement of points already there.
373  qreal rightMoveProp = 1.0f/(nColumns()-2) - 1.0f/(nColumns()-1);
374 
375  // Process all rows.
376  int k = 0;
377  for (int y=0; y<nRows(); y++)
378  {
379  // Get left and right vertices.
380  QPointF left = getVertex2d( 0, y );
381  QPointF right = getVertex2d( nColumns()-1, y );
382  QPointF diff = right - left;
383 
384  // Move all columns.
385  for (int x=0; x<nColumns(); x++)
386  {
387  // Ignore points from target column.
388  if (x == columnId)
389  continue;
390 
391  // Get current vertex.
392  QPointF p = getVertex2d( x, y );
393 
394  // The x value of this point in the new space.
395  int newX = x < columnId ? x : x-1;
396 
397  // Move middle points.
398  if (x > 0 && x < nColumns()-1)
399  {
400  p += (x < columnId ? +1 : -1) * diff * newX * rightMoveProp;
401  }
402 
403  // Assign new containers.
404  newVertices[k] = p;
405  newVertices2d[newX][y] = k;
406  k++;
407  }
408  }
409 
410  // Copy new mapping.
411  vertices = newVertices;
412  _vertices2d = newVertices2d;
413 
414  // Decrement number of columns.
415  _nColumns--;
416 
417  // Reorder.
418  _reorderVertices();
419 }

618 void Ellipse::setVertex(int i, const QPointF& v)
619 {
620  // Save vertical axis vector.
621  const QVector2D& vAxis = getVerticalAxis();
622 
623  // If changed one of the two rotation-controlling points, adjust the other two points.
624  if (i == 0 || i == 2)
625  {
626  // Transformation ellipse_t --> circle.
627  QTransform transform = toUnitCircle();
628 
629  // Change the vertex.
630  _rawSetVertex(i, v);
631 
632  // Combine with transformation circle -> ellipse_{t+1}.
633  transform *= fromUnitCircle();
634 
635  // Set vertices.
636  MShape::setVertex(1, transform.map( getVertex(1) ));
637  MShape::setVertex(3, transform.map( getVertex(3) ));
638  if (hasCenterControl())
639  MShape::setVertex(4, transform.map( getVertex(4) ));
640  }
641 
642  // If changed one of the two other points, just change the vertical axis.
643  else if (i == 1 || i == 3)
644  {
645  // Retrieve the new horizontal axis vector and center.
646  const QVector2D center(getCenter());
647 
648  QVector2D vFromCenter = QVector2D(v) - center;
649 
650  // Find projection of v onto vAxis / 2.
651  QVector2D vAxisNormalized = vAxis.normalized();
652  const QVector2D& projection = QVector2D::dotProduct( vFromCenter, vAxisNormalized ) * vAxisNormalized;
653 
654  // Assign vertical control points.
655  QPointF v1;
656  QPointF v3;
657  if (i == 1)
658  {
659  v1 = (center + projection).toPointF();
660  v3 = (center - projection).toPointF();
661  }
662  else
663  {
664  v1 = (center - projection).toPointF();
665  v3 = (center + projection).toPointF();
666  }
667 
668  // Transformation ellipse_t --> circle.
669  QTransform transform = toUnitCircle();
670 
671  // Change vertical points.
672  _rawSetVertex(1, v1);
673  _rawSetVertex(3, v3);
674 
675  // Combine with transformation circle -> ellipse_{t+1}.
676  transform *= fromUnitCircle();
677 
678  // Set vertices.
679  if (hasCenterControl())
680  _rawSetVertex(4, transform.map( getVertex(4) ));
681  }
682 
683  // Center control point (make sure it stays inside!).
684  else if (hasCenterControl())
685  {
686  // Clip control point.
687  _rawSetVertex(4, clipInside(v));
688  }
