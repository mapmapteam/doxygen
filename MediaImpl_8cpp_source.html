160 GstFlowReturn MediaImpl::gstNewSampleCallback(GstElement*, MediaImpl *p)
161 {
162  // Make it thread-safe.
163  p->lockMutex();
164 
165  // Get next frame.
166  GstSample *sample = gst_app_sink_pull_sample(GST_APP_SINK(p->_appsink0));
167 
168  // Unref last frame.
169  p->_freeCurrentSample();
170 
171  // Set current frame.
172  p->_currentFrameSample = sample;
173 
174  // For live sources, video dimensions have not been set, because
175  // gstPadAddedCallback is never called. Fix dimensions from first sample /
176  // caps we receive.
177  if (p->_isSharedMemorySource && ( p->_padHandlerData.width == -1 ||
178  p->_padHandlerData.height == -1)) {
179  GstCaps *caps = gst_sample_get_caps(sample);
180  GstStructure *structure;
181  structure = gst_caps_get_structure(caps, 0);
182  gst_structure_get_int(structure, "width", &p->_padHandlerData.width);
183  gst_structure_get_int(structure, "height", &p->_padHandlerData.height);
184  // g_print("Size is %u x %u\n", _padHandlerData.width, _padHandlerData.height);
185  }
186 
187  // Try to retrieve data bits of frame.
188  GstMapInfo& map = p->_mapInfo;
189  GstBuffer *buffer = gst_sample_get_buffer( sample );
190  if (gst_buffer_map(buffer, &map, GST_MAP_READ))
191  {
192  p->_currentFrameBuffer = buffer;
193  // For debugging:
194  //gst_util_dump_mem(map.data, map.size)
195 
196  // Retrieve data from map info.
197  p->_data = map.data;
198 
199  // Bits have changed.
200  p->_bitsChanged = true;
201  }
202 
203  p->unlockMutex();
204 
205  return GST_FLOW_OK;
206 }

632 void MediaImpl::_checkMessages()
633 {
634  if (_bus != NULL)
635  {
636  // Get message.
637  GstMessage *msg = gst_bus_timed_pop_filtered(
638  _bus, 0,
639  (GstMessageType) (GST_MESSAGE_STATE_CHANGED | GST_MESSAGE_ERROR | GST_MESSAGE_EOS | GST_MESSAGE_ASYNC_DONE));
640 
641  if (msg != NULL)
642  {
643  GError *err;
644  gchar *debug_info;
645 
646  switch (GST_MESSAGE_TYPE (msg))
647  {
648  // Error ////////////////////////////////////////////////
649  case GST_MESSAGE_ERROR:
650  gst_message_parse_error(msg, &err, &debug_info);
651  g_printerr("Error received from element %s: %s\n",
652  GST_OBJECT_NAME (msg->src), err->message);
653  g_printerr("Debugging information: %s\n",
654  debug_info ? debug_info : "none");
655  g_clear_error(&err);
656  g_free(debug_info);
657 
658  if (!_isSharedMemorySource)
659  {
660  _terminate = true;
661  }
662  else
663  {
664  _attached = false;
665  gst_element_set_state (_pipeline, GST_STATE_PAUSED);
666  gst_element_set_state (_pipeline, GST_STATE_NULL);
667  gst_element_set_state (_pipeline, GST_STATE_READY);
668  }
669 // _finish();
670  break;
671 
672  // End-of-stream ////////////////////////////////////////
673  case GST_MESSAGE_EOS:
674  // Automatically loop back.
675  g_print("End-Of-Stream reached.\n");
676  resetMovie();
677 // _terminate = true;
678 // _finish();
679  break;
680 
681  // Pipeline has prerolled/ready to play ///////////////
682  case GST_MESSAGE_ASYNC_DONE:
683  if (!_isMovieReady())
684  {
685  // Check if seeking is allowed.
686  gint64 start, end;
687  GstQuery *query = gst_query_new_seeking (GST_FORMAT_TIME);
688  if (gst_element_query (_pipeline, query))
689  {
690  gst_query_parse_seeking (query, NULL, (gboolean*)&_seekEnabled, &start, &end);
691  if (_seekEnabled)
692  {
693  g_print ("Seeking is ENABLED from %" GST_TIME_FORMAT " to %" GST_TIME_FORMAT "\n",
694  GST_TIME_ARGS (start), GST_TIME_ARGS (end));
695 
696  }
697  else
698  {
699  g_print ("Seeking is DISABLED for this stream.\n");

709  // Movie is ready!
710  _setMovieReady(true);
711  }

713  break;
714 
715  case GST_MESSAGE_STATE_CHANGED:
716  // We are only interested in state-changed messages from the pipeline.
717  if (GST_MESSAGE_SRC (msg) == GST_OBJECT (_pipeline))
718  {
719  GstState oldState, newState, pendingState;
720  gst_message_parse_state_changed(msg, &oldState, &newState,
721  &pendingState);

723  _uri.toUtf8().constData(),
724  gst_element_state_get_name(oldState),
725  gst_element_state_get_name(newState));
726  }
727  break;
728 
729  default:
730  // We should not reach here.
731  g_printerr("Unexpected message received.\n");
732  break;
733  }
734  gst_message_unref(msg);

751 {
752  if (_pipeline == NULL)
753  {
754  qDebug() << "Cannot set rate: no pipeline!" << endl;
755  return;
756  }
757 
758  if (!_seekEnabled)
759  {
760  qDebug() << "Cannot set rate: seek not working" << endl;
761  return;
762  }
763 
764  if (!_isMovieReady())
765  {
766  qDebug() << "Movie is not yet ready to play, cannot seek yet." << endl;
767  }
768 
769  gint64 position;
770  GstEvent *seekEvent;
771 
772  /* Obtain the current position, needed for the seek event */
773  if (!gst_element_query_position (_pipeline, GST_FORMAT_TIME, &position)) {
774  g_printerr ("Unable to retrieve current position.\n");
775  return;
776  }
777 
778  /* Create the seek event */
779  if (_rate > 0) {
780  seekEvent = gst_event_new_seek (_rate, GST_FORMAT_TIME, GstSeekFlags( GST_SEEK_FLAG_FLUSH | GST_SEEK_FLAG_ACCURATE ),
781  GST_SEEK_TYPE_SET, position, GST_SEEK_TYPE_NONE, 0);
782  } else {
783  seekEvent = gst_event_new_seek (_rate, GST_FORMAT_TIME, GstSeekFlags( GST_SEEK_FLAG_FLUSH | GST_SEEK_FLAG_ACCURATE ),
784  GST_SEEK_TYPE_SET, 0, GST_SEEK_TYPE_SET, position);
785  }
786 
787  if (_appsink0 == NULL) {
788  /* If we have not done so, obtain the sink through which we will send the seek events */
789  g_object_get (_pipeline, "video-sink", &_appsink0, NULL);
790  }

793  gst_element_send_event (_appsink0, seekEvent);

817 void MediaImpl::gstPadAddedCallback(GstElement *src, GstPad *newPad, MediaImpl::GstPadHandlerData* data)
818 {
819  g_print ("Received new pad '%s' from '%s':\n", GST_PAD_NAME (newPad), GST_ELEMENT_NAME (src));
820  GstPad *sinkPad = NULL;
821 
822  // Check the new pad's type.
823  GstCaps *newPadCaps = gst_pad_query_caps (newPad, NULL);
824  GstStructure *newPadStruct = gst_caps_get_structure (newPadCaps, 0);
825  const gchar *newPadType = gst_structure_get_name (newPadStruct);
826  gchar *newPadStructStr = gst_structure_to_string(newPadStruct);
827  g_print("Structure is %s\n", newPadStructStr);
828  g_free(newPadStructStr);
829  if (g_str_has_prefix (newPadType, "video/x-raw"))
830  {
831  sinkPad = gst_element_get_static_pad (data->videoToConnect, "sink");
832  gst_structure_get_int(newPadStruct, "width", &data->width);
833  gst_structure_get_int(newPadStruct, "height", &data->height);
834  }
835  else if (g_str_has_prefix (newPadType, "audio/x-raw"))
836  {
837  sinkPad = gst_element_get_static_pad (data->audioToConnect, "sink");
838  }
839  else {
840  g_print (" It has type '%s' which is not raw audio/video. Ignoring.\n", newPadType);
841  goto exit;
842  }
843 
844  // If our converter is already linked, we have nothing to do here.
845  if (gst_pad_is_linked (sinkPad))
846  {
847  // Best prefixes.
848  if (g_str_has_prefix (newPadType, "audio/x-raw-float") ||
849  g_str_has_prefix (newPadType, "video/x-raw-int") )
850  {
851  g_print (" Found a better pad.\n");
852  GstPad* oldPad = gst_pad_get_peer(sinkPad);
853  gst_pad_unlink(oldPad, sinkPad);
854  g_object_unref(oldPad);
855  }
856  else
857  {
858  g_print (" We are already linked. Ignoring.\n");
859  goto exit;
860  }
861  }
862 
863  // Attempt the link
864  if (GST_PAD_LINK_FAILED (gst_pad_link (newPad, sinkPad)))
865  {
866  g_print (" Type is '%s' but link failed.\n", newPadType);
867  goto exit;
868  } else {
869  data->videoIsConnected = true;
870  g_print (" Link succeeded (type '%s').\n", newPadType);
871  }
872 
873 exit:
874  // Unreference the new pad's caps, if we got them.
875  if (newPadCaps != NULL)
876  {
877  gst_caps_unref (newPadCaps);
878  }
879 
880  // Unreference the sink pad.
881  if (sinkPad != NULL)
882  {
883  gst_object_unref (sinkPad);
884  }
885 }
