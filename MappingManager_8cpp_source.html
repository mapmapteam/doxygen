122 QVector<Mapping::ptr> MappingManager::getVisibleMappings() const
123 {
124  QVector<Mapping::ptr> visible;
125 
126  // First pass: check if one of the mappings is in solo mode.
127  bool hasSolo = false;
128  for (QVector<Mapping::ptr>::const_iterator it = mappingVector.begin(); it != mappingVector.end(); ++it)
129  {
130  if ((*it)->isSolo())
131  {
132  hasSolo = true;
133  break;
134  }
135  }
136 
137  // Second pass: fill the visible vector.
138  for (QVector<Mapping::ptr>::const_iterator it = mappingVector.begin(); it != mappingVector.end(); ++it)
139  {
140  // Solo has priority over invisible (mute)
141  if ( (hasSolo && (*it)->isSolo() && (*it)->isVisible()) || (!hasSolo && (*it)->isVisible()) )
142  visible.push_back(*it);
143  }
144 
145  return visible;
146 }

148 void MappingManager::reorderMappings(QVector<uid> mappingIds)
149 {
150  // Both vector needs to have the same size.
151  Q_ASSERT( mappingIds.size() == mappingVector.size() );
152  mappingVector.clear();
153  int depth = 0;
154  for (QVector<uid>::iterator it = mappingIds.begin(); it != mappingIds.end(); ++it)
155  {
156  // Uid should be a key of the mappingMap.
157  Q_ASSERT( mappingMap.contains(*it) );
158  // Makes sure the uids are not repeated.
159  Q_ASSERT( !mappingVector.contains(mappingMap[*it]) );
160  // Adds the mapping at the right place in the vector.
161  Mapping::ptr mapping = mappingMap[*it];
162  mapping->setDepth(depth);
163  mappingVector.push_back( mapping );
164 
165  depth++;
166  }
167 }
