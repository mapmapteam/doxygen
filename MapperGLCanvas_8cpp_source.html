103 void MapperGLCanvas::mousePressEvent(QMouseEvent* event)
104 {
105  bool mousePressedOnSomething = false;
106 
107  _mousePressedPosition = event->pos();
108  QPointF pos = mapToScene(event->pos());
109 
110  // Drag the scene with middle button.
111  if (event->buttons() & Qt::MiddleButton)
112  {
113  // NOTE: This is a trick code to implement scroll hand drag using the middle button.
114  QMouseEvent releaseEvent(QEvent::MouseButtonRelease, event->pos(), event->globalPos(), Qt::LeftButton, 0, event->modifiers());
115  QGraphicsView::mouseReleaseEvent(&releaseEvent);
116  setDragMode(QGraphicsView::ScrollHandDrag);
117 
118  // We need to pretend it is actually the left button that was pressed!
119  QMouseEvent fakeEvent(event->type(), event->pos(), event->globalPos(),
120  Qt::LeftButton, event->buttons() | Qt::LeftButton, event->modifiers());
121  QGraphicsView::mousePressEvent(&fakeEvent);
122  }
123 
124  // Check for vertex selection first.
125  else if (event->buttons() & Qt::LeftButton)
126  {
127  MShape::ptr shape = getCurrentShape();
128  if (shape)
129  {
130  // Note: we compare with the square value for fastest computation of the distance
131  int minDistance = sq(MM::VERTEX_SELECT_RADIUS);
132 
133  // Find the ID of the nearest vertex (from currently selected shape)
134  for (int i = 0; i < shape->nVertices(); i++)
135  {
136  int dist = distSq(_mousePressedPosition, mapFromScene(shape->getVertex(i))); // squared distance
137  if (dist < minDistance)
138  {
139  _activeVertex = i;
140  minDistance = dist;
141 
142  _vertexGrabbed = true;
143  mousePressedOnSomething = true;
144 
145  _grabbedObjectStartScenePosition = shape->getVertex(i);
146  }
147  }
148  }
149  }
150 
151  if (mousePressedOnSomething)
152  return;
153 
154  // Check for shape selection.
155  if (event->buttons() & (Qt::LeftButton | Qt::RightButton)) // Add Right click for context menu
156  {
157  MShape::ptr selectedShape = getCurrentShape();
158 
159  // Possibility of changing shape in output by clicking on it.
160  MappingManager manager = getMainWindow()->getMappingManager();
161  QVector<Mapping::ptr> mappings = manager.getVisibleMappings();
162  for (QVector<Mapping::ptr>::const_iterator it = mappings.end() - 1; it >= mappings.begin(); --it)
163  {
164  MShape::ptr shape = getShapeFromMappingId((*it)->getId());
165 
166  // Check if mouse was pressed on that shape.
167  if (shape && shape->includesPoint(pos))
168  {
169  mousePressedOnSomething = true;
170 
171  // Deselect vertices.
172  deselectVertices();
173 
174  // Change mapping (only available in destination).
175  if (isOutput() && shape != selectedShape)
176  {
177  // Change current mapping.
178  getMainWindow()->setCurrentMapping((*it)->getId());
179 
180  // Reset selected shape to new one.
181  selectedShape = getCurrentShape();
182  }
183 
184  break;
185  }
186  }
187 
188  // Grab the shape.
189  if (event->buttons() & Qt::LeftButton) // This preserve me from duplicate code above
190  {
191  if (selectedShape && selectedShape->includesPoint(pos))
192  {
193  _shapeGrabbed = true;
194  _shapeFirstGrab = true;
195 
196  _grabbedObjectStartScenePosition = pos;
197  }
198  }
199  }
200 
201  if (mousePressedOnSomething)
202  return;
203 
204  // Deactivate.
205  deselectAll();
206 }

467 void MapperGLCanvas::wheelEvent(QWheelEvent *event)
468 {
469  int deltaLevel = event->delta() / 120;
470  qreal zoomFactor = qPow(MM::ZOOM_FACTOR, _zoomLevel);
471  if (deltaLevel > 0)
472  {
473  // First check if we're already at max.
474  while (deltaLevel && zoomFactor < MM::ZOOM_MAX) {
475  _zoomLevel++;
476  deltaLevel--;
477  zoomFactor = qPow(MM::ZOOM_FACTOR, _zoomLevel);
478  }
479  zoomFactor = qMin(zoomFactor, MM::ZOOM_MAX);
480  }
481  else
482  {
483  // First check if we're already at min.
484  while (deltaLevel && zoomFactor > MM::ZOOM_MIN) {
485  _zoomLevel--;
486  deltaLevel++;
487  zoomFactor = qPow(MM::ZOOM_FACTOR, _zoomLevel);
488  }
489  zoomFactor = qMax(zoomFactor, MM::ZOOM_MIN);
490  }
491 
492  // Re-bound zoom (for consistency).
493  zoomFactor = getZoomFactor();
494 
495  // Apply zoom to view.
496  QGraphicsView* view = scene()->views().first();
497  view->resetMatrix();
498  view->scale(zoomFactor, zoomFactor);
499  view->update();
500 
501  // Accept wheel scrolling event.
502  event->accept();
503 }
