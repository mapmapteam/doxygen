67 void MappingGui::setValue(QtProperty* property, const QVariant& value)
68 {
69  if (property == _opacityItem)
70  {
71  double opacity = qBound(value.toDouble() / 100.0, 0.0, 1.0);
72  if (opacity != _mapping->getRawOpacity())
73  {
74  _mapping->setRawOpacity(opacity);
75  emit valueChanged();
76  }
77  }
78  else
79  {
80  std::map<QtProperty*, std::pair<MShape*, int> >::iterator it = _propertyToVertex.find(property);
81  if (it != _propertyToVertex.end())
82  {
83  const QPointF& p = value.toPointF();
84  MShape* shape = it->second.first;
85  int v = it->second.second;
86  if (shape->getVertex(v) != p)
87  {
88  shape->setVertex(v, p);
89  emit valueChanged();
90  }
91  }
92  }
93 }

103 void MappingGui::_buildShapeProperty(QtProperty* shapeItem, MShape* shape)
104 {
105  for (int i=0; i<shape->nVertices(); i++)
106  {
107  // Add point.
108  QtVariantProperty* pointItem = _variantManager->addProperty(QVariant::PointF,
109  QObject::tr("Point %1").arg(i));
110 
111  const QPointF& p = shape->getVertex(i);
112  pointItem->setValue(p);
113 
114  shapeItem->addSubProperty(pointItem);
115  _propertyToVertex[pointItem] = std::make_pair(shape, i);
116  }
