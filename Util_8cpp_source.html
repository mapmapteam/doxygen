178 void drawControlsEllipse(QPainter* painter, const QList<int>* selectedVertices, const Ellipse& ellipse)
179 {
180  // Init colors and stroke.
181  painter->setPen(MM::SHAPE_STROKE);
182  painter->setBrush(Qt::NoBrush);
183 
184  // Draw ellipse contour.
185  qreal rotation = ellipse.getRotation();
186 
187  painter->save(); // save painter state
188 
189  painter->resetTransform();
190  const QPointF& center = ellipse.getCenter();
191  painter->translate(center);
192  painter->rotate(rotation);
193  painter->drawEllipse(QPointF(0,0), ellipse.getHorizontalRadius(), ellipse.getVerticalRadius());
194 
195  painter->restore(); // restore saved painter state
196 
197  // Draw control points.
