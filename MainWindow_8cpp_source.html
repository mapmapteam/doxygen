283 bool MainWindow::eventFilter(QObject *obj, QEvent *event)
284 {
285  bool eventKey = false;
286  if (event->type() == QEvent::KeyPress)
287  {
288  QKeyEvent *keyEvent = static_cast<QKeyEvent *>(event);
289  eventKey = true;
290  // Menubar shortcut
291  if (keyEvent->modifiers() == Qt::CTRL)
292  {
293  switch (keyEvent->key()) {
294  case Qt::Key_F:
295  if (outputWindow->windowState() != Qt::WindowFullScreen)
296  {
297  outputWindow->setFullScreen(true);
298  outputWindowFullScreenAction->setChecked(true);
299  outputWindowFullScreenAction->setEnabled(true);
300  }
301  break;
302  case Qt::Key_N:
303  newFile();
304  break;
305  case Qt::Key_O:
306  open();
307  break;
308  case Qt::Key_S:
309  save();
310  break;
311  case Qt::Key_Q:
312  close();
313  break;
314  case Qt::Key_Delete:
315  deleteItem();
316  break;
317  case Qt::Key_M:
318  addMesh();
319  break;
320  case Qt::Key_T:
321  addTriangle();
322  break;
323  case Qt::Key_E:
324  addEllipse();
325  break;
326  case Qt::Key_W:
327  outputWindow->setVisible(true);
328  break;
329  case Qt::Key_R:
330  rewind();
331  break;
332  case Qt::Key_Z:
333  undoStack->undo();
334  break;
335  }
336  }
337  else if (keyEvent->matches(QKeySequence::Redo))
338  {
339  undoStack->redo();
340  }
341  else if (keyEvent->key() == Qt::Key_Escape)
342  {
343  outputWindow->setFullScreen(false);
344  outputWindowFullScreenAction->setChecked(false);
345  outputWindowFullScreenAction->setEnabled(false);
346  }
347  else if (keyEvent->key() == Qt::Key_Space)
348  {
349  if (_isPlaying)
350  {
351  pause();
352  }
353  else
354  {
355  play();
356  }
357  }
358  eventKey = false;
359 
360  return eventKey;
361  }
362  else
363  {
364  // standard event processing
365  return QObject::eventFilter(obj, event);
366  }
