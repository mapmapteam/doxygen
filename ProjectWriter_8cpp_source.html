140 void ProjectWriter::writeItem(Mapping *item)
141 {
142  _xml.writeStartElement("mapping");

144  _xml.writeAttribute("id", QString::number(item->getId()));
145  _xml.writeAttribute("paint_id", QString::number(item->getPaint()->getId()));
146  _xml.writeAttribute("type", item->getType());
147 
148  // boolean attributes:
149  _xml.writeAttribute("locked", QString::number((int) item->isLocked() ? 1 : 0));
150  _xml.writeAttribute("solo", QString::number((int) item->isSolo() ? 1 : 0));
151  _xml.writeAttribute("visible", QString::number((int) item->isVisible() ? 1 : 0));
152 
153  MShape *shape = item->getShape().data();
154  _xml.writeStartElement("destination");
155  _xml.writeAttribute("shape", shape->getType());
156  writeShapeVertices(shape);
157  _xml.writeEndElement(); // shape
158 
159  if (item->getType().endsWith("_texture"))
160  {
161  TextureMapping *tex = (TextureMapping *) item;
162  shape = tex->getInputShape().data();
163 
164  _xml.writeStartElement("source");
165  _xml.writeAttribute("shape", shape->getType());
166  writeShapeVertices(shape);
167  _xml.writeEndElement(); // shape
168 
169  _xml.writeEndElement(); // mapping
170  }
171  else if (item->getType().endsWith("_color"))
172  {
173  _xml.writeEndElement(); // mapping
174  }
175  else
176  qDebug() << "Unknown type, cannot save: " << item->getType() << endl;
177 }
